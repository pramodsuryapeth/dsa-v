function mergeSort(array)
    if array has 1 element
        return array
    split array into two halves
    left = mergeSort(left half)
    right = mergeSort(right half)
    return merge(left, right)

function merge(left, right)
    while left and right have elements
        if left[0] < right[0]
            push left[0] to result and remove from left
        else
            push right[0] to result and remove from right
    push remaining elements of left and right to result
    return result
