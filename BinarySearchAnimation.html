Pseudocode for Binary Search:

procedure BinarySearch(array A, target)
    low = 0
    high = length of A - 1
    while low <= high do
        mid = (low + high) / 2
        if A[mid] == target then
            return mid
        else if A[mid] < target then
            low = mid + 1
        else
            high = mid - 1
    end while
    return -1
end procedure
