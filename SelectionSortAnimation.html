Pseudocode for Selection Sort:

procedure SelectionSort(array A)
    for i ← 0 to length(A) - 1 do
        minIndex = i
        for j ← i + 1 to length(A) do
            if A[j] < A[minIndex] then
                minIndex = j
            end if
        end for
        swap A[i] and A[minIndex]
    end for
end procedure
