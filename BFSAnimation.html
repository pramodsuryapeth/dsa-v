FUNCTION BFS(treeRoot)
    INITIALIZE queue as empty list
    INITIALIZE visited as empty set

    ENQUEUE treeRoot into queue
    ADD treeRoot to visited

    WHILE queue is not empty
        DEQUEUE node from queue as currentNode
        PRINT "Processing: currentNode"

        FOR EACH child in currentNode's children
            IF child is not in visited
                ENQUEUE child into queue
                ADD child to visited
            END IF
        END FOR

        PRINT "Queue: [elements in queue]"
    END WHILE

    PRINT "BFS complete!"
END FUNCTION
