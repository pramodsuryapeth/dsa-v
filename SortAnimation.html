Pseudocode for Bubble Sort:

procedure BubbleSort(array A, n)
    for i from 0 to n-1 do
        for j from 0 to n-i-2 do
            if A[j] > A[j+1] then
                // Swap the elements
                temp ← A[j]
                A[j] ← A[j+1]
                A[j+1] ← temp
            end if
        end for
    end for
end procedure
