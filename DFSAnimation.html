FUNCTION DFS(treeRoot)
    INITIALIZE stack as empty list
    INITIALIZE visited as empty set

    PUSH treeRoot into stack
    ADD treeRoot to visited

    WHILE stack is not empty
        POP node from stack as currentNode
        PRINT "Processing: currentNode"

        FOR EACH child in currentNode's children
            IF child is not in visited
                PUSH child into stack
                ADD child to visited
            END IF
        END FOR

        PRINT "Stack: [elements in stack]"
    END WHILE

    PRINT "DFS complete!"
END FUNCTION
